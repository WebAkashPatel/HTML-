PHP Code Hacks - 31 to 50

31. Recursive Functions


function factorial($n) {
    return $n <= 1 ? 1 : $n * factorial($n - 1);
}
echo factorial(5); // Outputs: 120

32. Read Directory Contents


$files = scandir('.');
print_r($files); // List of files in the current directory

33. Simple Array Filtering


$nums = [1, 2, 3, 4, 5];
$even = array_filter($nums, fn($n) => $n % 2 === 0);
print_r($even); // Outputs: [2, 4]

34. Convert Object to Array


$obj = (object) ['a' => 1, 'b' => 2];

print_r($arr); // Outputs: ['a' => 1, 'b' => 2]

38. Convert String to Array


$string = 'one,two,three';
$array = explode(',', $string);
print_r($array); // Outputs: ['one', 'two', 'three']

39. Array to String


$array = ['one', 'two', 'three'];
$string = implode(',', $array);
echo $string; // Outputs: one,two,three

40. Use Spread Operator for Arrays


$array1 = [1, 2];
$array2 = [...$array1, 3, 4];
print_r($array2); // Outputs: [1, 2, 3, 4]

41. Array Unique Values

43. Get Array Keys

44. Get Array Values

45. Sort Array by Values

46. Sort Array by Keys


$array = ['b' => 2, 'a' => 1];

47. Create Range of Numbers

48. Array Combine


$keys = ['a', 'b'];
$values = [1, 2];
$combined = array_combine($keys, $values);
print_r($combined); // Outputs: ['a' => 1, 'b' => 2]
